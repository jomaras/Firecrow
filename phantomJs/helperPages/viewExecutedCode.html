<script>
var externalObj =
{}
;
var mouseDownPosition;
var firstContainer = document.getElementById("firstContainer");
var position;
firstContainer.onmousemove = function (eventArgs)
{
if (mouseDownPosition == null)
{
return mouseDownPosition;
}
if (eventArgs.pageX < mouseDownPosition.x && eventArgs.pageY < mouseDownPosition.y)
{
firstContainer.textContent = "Is Dragging Left Up"
}
else
if (eventArgs.pageX >= mouseDownPosition.x && eventArgs.pageY < mouseDownPosition.y)
{
firstContainer.textContent = "Is Dragging right Up"
}
else
if (eventArgs.pageX >= mouseDownPosition.x && eventArgs.pageY >= mouseDownPosition.y)
{
firstContainer.textContent = "Is Dragging right down"
}
else
if (eventArgs.pageX < mouseDownPosition.x && eventArgs.pageY >= mouseDownPosition.y)
{
firstContainer.textContent = "Is Dragging Left down"
}
}
firstContainer.onmousedown = function (eventArgs)
{
mouseDownPosition =
{
x: eventArgs.pageX,
y: eventArgs.pageY
}
}
firstContainer.onmouseup = function ()
{
mouseDownPosition = null
}
</script>
